select count(distinct Instructor_id) as inst_each_dept,d.name from instructor i join department d on i.dept_id = d.dept_id group by d.name;

2. display departments having dallas, new York, and Chicago location:
select dept_name from departments where location = 'dallas' or location = 'new york' or location = 'Chicago';
OR
select dept_name from departments where lower(location) in ('dallas','new york','chicago');

3. display the names of employees who are working as clerk, salesman, or analyst and drawing a salary of more than 3000
select first_name, job, salary from emp1;
select first_name, job, salary from emp1 where salary>=3000 and lower(job) in ('clerk','salesman','analyst');

4. display the names of employees who are not working as manager.
select first_name, last_name from emp1 where lower(job) != 'manager';

5. display the name of the employee who earns the highest salary.
select max(salary) from emp1;
select first_name, salary from emp1 where salary = (select max(salary) from emp1));

6. display he employee number and name of employee working as clerk and earning highest salary among clerks.
