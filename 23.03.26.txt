program:
create or replace procedure pr1(id in Number)
AS
name emp.empname %type;
salary emp.salary %type;
begin
select empname, salary into Name, V_Salary from emp
where emp_id = id;
dbms_output.put_line('Name : '|| name); 
dbms_output.put_line('Salary : '|| V_Salary);
exception when no_data found then
dbms.output.put_line('id not found');
end pr1;
